import {
  Accordion as ConnectorAccordion,
  AccordionSummary as ConnectorAccordionSummary,
  AccordionDetails as ConnectorAccordionDetails,
} from '@site/src/components/ConnectorAccordion';

import {Accordion as ReleaseAccordion} from '@site/src/components/ReleaseAccordion';

import { useEffect, useState } from "react"

export const Connectors = ({url}) => {
  const [connectors, setConnectors] = useState([])
  const fetchConnectorData = () => {
    fetch(url)
      .then(response => {
        return response.json()
      })
      .then(data => {
        setConnectors(data)
      })
  }
  useEffect(() => {
    fetchConnectorData()
  }, [])
  return (
    <div className="connectors">
      {connectors.map(conn => (
        <ConnectorAccordion>
          <ConnectorAccordionSummary
            name={conn.nameWithOwner}
            url={conn.url}
            stargazerCount={conn.stargazerCount}
            description={conn.description}
          />
          <ConnectorAccordionDetails>
            {conn.releases.map(release => (
              <ReleaseAccordion release={release} />
            ))}
          </ConnectorAccordionDetails>
        </ConnectorAccordion>
      ))}
    </div>
  )
};
