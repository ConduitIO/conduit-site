import {
  Accordion,
  AccordionSummary,
  AccordionDetails,
  ReleaseAccordion,
  ReleaseAccordionSummary,
  ReleaseAccordionDetails,
} from '@site/src/components/ConnectorsAccordion';

import { useEffect, useState } from "react"

export const Connectors = ({url}) => {
  const [connectors, setConnectors] = useState([])
  const fetchConnectorData = () => {
    fetch(url)
      .then(response => {
        return response.json()
      })
      .then(data => {
        setConnectors(data)
      })
  }
  useEffect(() => {
    fetchConnectorData()
  }, [])
  return (
    <div className="connectors">
      {connectors.map(conn => (
        <Accordion>
          <AccordionSummary
            name={conn.nameWithOwner}
            url={conn.url}
            stargazerCount={conn.stargazerCount}
            description={conn.description}
          />
          <AccordionDetails>
            {conn.releases.map(release => (
              <ReleaseAccordion>
                <ReleaseAccordionSummary
                  tag={release.tag_name}
                />
                <ReleaseAccordionDetails>
                  <ul className="mb-4">
                    {release.body.split("\n").map(line => (
                      <li key={line}>{line}</li>
                    ))}
                    <li>Downloads:
                      <ul>
                        {release.assets.map(asset => (
                          <li key={asset.name}>{asset.os}/{asset.arch}: <a href={asset.browser_download_url}>{asset.name}</a></li>
                        ))}
                      </ul>
                    </li>
                  </ul>
                </ReleaseAccordionDetails>
              </ReleaseAccordion>
            ))}
          </AccordionDetails>
        </Accordion>
      ))}
    </div>
  )
};
