import ConnectorAccordion from '@site/src/components/ConnectorAccordion';

import { useEffect, useState } from "react"

export const Connectors = ({url}) => {
  const [connectors, setConnectors] = useState([])
  const fetchConnectorData = () => {
    fetch(url)
      .then(response => {
        return response.json()
      })
      .then(data => {
        setConnectors(data)
      })
  }
  useEffect(() => {
    fetchConnectorData()
  }, [])
  return (
    <>
      {connectors.map(conn => (
        <ConnectorAccordion connector={conn} />
      ))}
    </>
  )
};
